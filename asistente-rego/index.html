async function processAudio(blob) { const formData = new FormData(); formData.append('audio', blob); try { // USAR URL ABSOLUTA PARA EVITAR PROBLEMAS DE RUTA const res = await fetch('https://alex-ra-sa.vercel.app/api/asistente', { method: 'POST', body: formData }); const data = await res.json(); if (data.success) { lastPlan = data.plan; resultBox.style.display = 'block'; document.getElementById('transcript').innerText = `"${data.transcript}"`; if (data.needSelection) { aiMessage.innerText = `Encontré ${data.accounts.length} cuentas. Selecciona la correcta:`; showAccountList(data.accounts); } else { aiMessage.innerText = "No encontré ninguna cuenta con ese nombre. Intenta pronunciar el nombre más claro."; } statusText.innerText = "Listo"; } else { statusText.innerText = "Error en el servidor"; aiMessage.innerText = "Error: " + data.error; resultBox.style.display = 'block'; } } catch (e) { statusText.innerText = "Fallo de red"; alert("Error de conexión: " + e.message); } }
